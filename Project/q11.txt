yaml:

apiVersion: v1
kind: Pod
metadata:
  name: redis-storage-yourname
spec:
  containers:
    - name: redis-container
      image: redis:alpine
      volumeMounts:
        - name: redis-storage
          mountPath: /data/redis
  volumes:
    - name: redis-storage
      emptyDir: {}



command: kubectl apply -f redis-storage-pod.yaml
